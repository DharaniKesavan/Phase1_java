Create table query:

create table eproduct(Id integer not null unique, name varchar(30) not null, price integer);
insert into eproduct values(1, "HP", 30000);
insert into eproduct values(2, "Apple", 45000);
insert into eproduct values(3, "Product3", 17800);

Stored Procedure query:

DELIMITER $$
CREATE PROCEDURE add_product(IN PID integer, IN pname varchar(30), IN pprice integer)
INSERT INTO eproduct (Id, name, price) VALUES (PID, pname, pprice)
$$
DELIMITER ;
